todos = []

while True:
    user_action = input("Type add or show: ").strip()

    match user_action:
        case "add":
            todo = input("Enter a todo item: ")
            todos.append(todo)
        case "show":
            for item in todos:
                print(item)
        case "exit":
            break
        case _:
            print("Invalid input")
print("Goodbye!")

-----------------

_: = no any match
